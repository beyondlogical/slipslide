# Installation

```shell
$ npm install --save rimble-ui styled-components

$ yarn add rimble-ui
```

---
# Usage

```javascript
import React, { Component } from 'react'
import { Button } from 'rimble-ui'
class Example extends Component {
  render () {
    return (
      <Button size={'medium'}>
        Click me!
      </Button>
    )
  }
}
```
